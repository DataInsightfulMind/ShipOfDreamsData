import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns

data=pd.read_csv("titanic.csv")
df1=pd.DataFrame(data)


df1['age'] = df1['age'].replace(np.nan, 30)
data['sex']=data['sex'].replace({'female':0,'male':1})
data['FamilySize'] = data['sibsp'] + data['parch'] + 1
mean_age = np.mean(data['age'].dropna())
print(mean_age)
median_age = np.median(data['age'].dropna())
print(median_age)
std_dev_age = np.std(data['age'].dropna())
print(std_dev_age)

most_frequent_embarked = data['embark_town'].mode()[0]
print(most_frequent_embarked)
print(df1,data)
data['pclass'].value_counts().plot(kind='bar')


plt.xlabel('pclass')
plt.ylabel('count')
plt.title('count of passengers by pclass')
plt.show()
plt.hist(data['age'],bins=20)
plt.xlabel("age")
plt.ylabel("frequency")
plt.plot('Histogram of age')
plt.show()
corr_matrix = data[['pclass', 'sex', 'age', 'survived']].corr()
plt.figure(figsize=(8, 6))
sns.heatmap(corr_matrix, annot=True, cmap='coolwarm', square=True)
plt.title('correlation between pclass, sex, age, and survived')
plt.show()
